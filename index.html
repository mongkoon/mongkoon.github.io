<script>

// Set the date we're counting down to

var limit = Number(getStoredValue("time_limit"));


function myTimer() {

    

    limit = Number(getStoredValue("time_limit"));

    

//   // Time calculations for days, hours, minutes and seconds

  var hours = Math.floor(limit / (1000 * 60 * 60));

  var minutes = Math.floor((limit % (1000 * 60 * 60)) / (1000 * 60));

  var seconds = Math.floor((limit % (1000 * 60)) / 1000);

    

//   // Output the result in an element with id="demo"

  document.getElementById("demo").innerHTML = hours + "시 " + minutes + "분 " + seconds + "초 ";


    // document.getElementById("demo").innerHTML = "test";

  

  limit = limit - 1000;

  storeValue('time_limit', String(limit));

}


// Update the count down every 1 second

var x = setInterval(myTimer, 1000);


var mode = getStoredValue('myPageMode');


function buttonClick(mode) {

    mode = mode;

    storeValue('myPageMode', mode);

}


function storeValue(key, value) {

    if (localStorage) {

        localStorage.setItem(key, value);

    } else {

        $.cookies.set(key, value);

    }

}


function getStoredValue(key) {

    if (localStorage) {

        return localStorage.getItem(key);

    } else {

        return $.cookies.get(key);

    }

}


function stopTimer()

{

  if (x !== -1) {

    clearInterval(x);

    x = -1; 

  }

  document.getElementById("stopBtn").innerHTML = "다시시작";

}


function startTimer()

{

  if (x === -1) {

    x = setInterval(myTimer, 1000);

  }

  document.getElementById("stopBtn").innerHTML = "일시정지";

}


function stopOrStart() 

{

    if (x !== -1) {

        stopTimer();

    } else {

        startTimer();

    }

}


</script>

<p id="demo"><button id="stopBtn" onclick = "stopOrStart();" > 일시정지 </button></p>
